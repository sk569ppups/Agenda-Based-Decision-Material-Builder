function buildInstruction(topic, decision, helpers, adds, excludes) { let text = `${topic}について調べてください。`; if (decision) { text += ` 最終的に、${decision}ための判断材料を整理したいです。`; } if (helpers.length) { text += ` ${helpers.join("、")}の観点を含めて整理してください。`; } if (adds.length) { text += ` ${adds.join("、")}${adds.length ? "。" : ""}`; } if (excludes.length) { text += ` ${excludes.join("、")}してください。`; } return text.trim(); } document.getElementById("nextButton").addEventListener("click", () => { const topic = document.getElementById("topic").value.trim(); const decision = document.getElementById("decision").value.trim(); const helpers = getCheckedValues("helperKeywords"); const adds = getCheckedValues("addConditions"); const excludes = getCheckedValues("excludeConditions"); if (!topic) { alert("『調べたいこと』を入力してください。"); document.getElementById("topic").focus(); return; } if (!decision) { alert("『決めたいこと』を入力してください。"); document.getElementById("decision").focus(); return; } const summaryArea = document.getElementById("summaryArea"); summaryArea.innerHTML = `
調べたいこと：${escapeHtml(topic)}
決めたいこと：${escapeHtml(decision)}
補助ワード：${escapeHtml(joinOrFallback(helpers))}
加えたい条件：${escapeHtml(joinOrFallback(adds))}
除外したい条件：${escapeHtml(joinOrFallback(excludes))}
`; const instruction = buildInstruction(topic, decision, helpers, adds, excludes); document.getElementById("instructionText").textContent = instruction; document.getElementById("resultSection").classList.remove("hidden"); document.getElementById("copyMessage").textContent = ""; document.getElementById("resultSection").scrollIntoView({ behavior: "smooth", block: "start" }); }); document.getElementById("copyButton").addEventListener("click", async () => { const text = document.getElementById("instructionText").textContent.trim(); if (!text) return; try { await navigator.clipboard.writeText(text); document.getElementById("copyMessage").textContent = "コピーしました。Perplexityへ貼り付けてご利用ください。"; } catch (error) { document.getElementById("copyMessage").textContent = "コピーに失敗しました。手動でコピーしてください。"; } }); document.getElementById("openPerplexityButton").addEventListener("click", () => { window.open("https://www.perplexity.ai/", "_blank", "noopener,noreferrer"); }); function escapeHtml(str) { return str .replace(/&/g, "&") .replace(//g, ">") .replace(/\"/g, """) .replace(/'/g, "'"); }
